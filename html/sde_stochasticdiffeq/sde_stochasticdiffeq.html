Stochastic differential equation model using StochasticDiffEq.jl

Simon Frost (@sdwfrost), 2020-04-27

Introduction

A stochastic differential equation version of the SIR model is:

  • Stochastic

  • Continuous in time

  • Continuous in state

This implementation uses StochasticDiffEq.jl, which has a variety of SDE solvers.

Transitions

We begin by specifying the ODE kernel.

function sir_ode!(du,u,p,t)
    (S,I,R) = u
    (β,c,γ) = p
    N = S+I+R
    @inbounds begin
        du[1] = -β*c*I/N*S
        du[2] = β*c*I/N*S - γ*I
        du[3] = γ*I
    end
    nothing
end;

function sir_noise!(du,u,p,t)
    (S,I,R) = u
    (β,c,γ) = p
    N = S+I+R
    ifrac = β*I/N*S
    rfrac = γ*I
    du[1,1] = -sqrt(ifrac)
    du[2,1] = sqrt(ifrac)
    du[2,2] = -sqrt(rfrac)
    du[3,2] = sqrt(rfrac)
end;
